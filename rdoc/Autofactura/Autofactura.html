def initialize(config = {})
  
  self.url = config[:url].blank? ? "http://app.autofactura.com/users/api/" : config[:url]
  self.user = config[:user].blank? ? "" : config[:user]
  self.sucursal = config[:sucursal].blank? ? "" : config[:sucursal]
  
end
